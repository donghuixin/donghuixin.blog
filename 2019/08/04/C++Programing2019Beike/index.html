2019贝壳找房秋招编程题

第一题，找出相邻的绝对值最小组

/*
输入

第一行 数字个数
    9

    -1
    1
    3
    7
    2
    4
    6
    111111111111110
    111111111111111

输出
3 4
*/

#include<algorithm>
#include<string>
#include<vector>
#include<iostream>
using namespace std;
class Solution {
public:
 void soulution(int n,vector<long>so)
 {
  long min = abs(so[0]-so[1]);
  long a, b;
  a = so[0]; b = so[1];
  for (int i = 1; i < n; i++)
  {
   if (abs(so[i] - so[i - 1]) < min)
   {
    a = so[i - 1];
    b = so[i];
    min = abs(so[i] - so[i - 1]);
   }
   else
    continue;
  }
  cout << a <<' '<< b << endl;
 }
};
int main()
{
 int n;
 cin >> n;
 vector<long>so;
 so.resize(n);
 for (int i = 0; i < n; i++)
  cin >> so[i];
 Solution s;
 s.soulution(n, so);

}
